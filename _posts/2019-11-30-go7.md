# struct

```go
type Student struct {
     Name string
     Age int
}

func main(){
     s1 := Student{}
     s1.Name="Lin"
     s1.Age=20
     fmt.Println(s1)

     s2 :=Student{
          Name:"Lee",
          Age:22,
     }
     fmt.Println(s2)

     //指向結構的指針. 傳入的參數更改到s3自身.
     s3 :=&Student{
          Name:"Wang",
          Age:22,
     }
     MakeAnStu(s3)
}

func MakeAnStu(s *Student){
     s.Age=99
}
```

#### 匿名結構
```go

func main(){
     a := &struct {
          Name string
          Age int
     }{
          Name: "joe",
          Age:19,
     }
}
```
